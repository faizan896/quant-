    def __init__(self):
        self.prices = {}
        self.history = {}
        random.seed(42)
        self._init()

    def _init(self):
        for tk, info in self.ASSETS.items():
            base = info["price"]; vol = info["vol"]; dv = vol/math.sqrt(252)
            p = base * 0.85; hist = []
            for i in range(252):
                drift = 0.0002 + random.gauss(0, dv)
                p *= (1 + drift)
                dt = datetime.now() - timedelta(days=252-i)
                o = p*(1+random.gauss(0,dv*0.3))
                h = max(o,p)*(1+abs(random.gauss(0,dv*0.5)))
                l = min(o,p)*(1-abs(random.gauss(0,dv*0.5)))
                hist.append({"date":dt.strftime("%Y-%m-%d"),"open":round(o,4),"high":round(h,4),"low":round(l,4),"close":round(p,4),"volume":int(random.uniform(5e6,50e6))})
            self.prices[tk] = round(p,4)
            self.history[tk] = hist

    def tick(self):
        mf = random.gauss(0,0.001); rf = random.gauss(0,0.0005)
        updates = {}
        for tk, info in self.ASSETS.items():
            dv = info["vol"]/math.sqrt(252)/math.sqrt(78)
            idio = random.gauss(0, dv)
            t = info["type"]
            if t in ("equity","etf","index"): ch = mf*0.6+rf*0.2+idio*0.5
            elif t == "commodity": ch = mf*0.2-rf*0.3+idio*0.7
            elif t == "bond": ch = -mf*0.3+idio*0.3
            else: ch = mf*0.1+idio*0.4
            old = self.prices[tk]; new = round(old*(1+ch),4)
            self.prices[tk] = new
            updates[tk] = {"ticker":tk,"price":new,"change":round(new-old,4),"changePct":round((new-old)/old*100,4),"name":info["name"],"type":info["type"]}
        return updates

# ── Quant Engine ────────────────────────────────────────────
class QuantEngine:
    @staticmethod
    def returns(prices):
        return [math.log(prices[i]/prices[i-1]) for i in range(1,len(prices)) if prices[i-1]>0]

    @staticmethod
    def ma(prices, w):
        r = []
        for i in range(len(prices)):
            if i < w-1: r.append(None)
            else: r.append(sum(prices[i-w+1:i+1])/w)
        return r

    @staticmethod
    def ema(prices, w):
        if not prices: return []
        k=2/(w+1); r=[prices[0]]
        for i in range(1,len(prices)): r.append(prices[i]*k+r[-1]*(1-k))
        return r

    @staticmethod
    def rsi(prices, period=14):
        if len(prices)<period+1: return [50]*len(prices)
        d=[prices[i]-prices[i-1] for i in range(1,len(prices))]
        g=[max(x,0) for x in d]; l=[-min(x,0) for x in d]
        ag=sum(g[:period])/period; al=sum(l[:period])/period
        r=[None]*period
        for i in range(period,len(d)):
            ag=(ag*(period-1)+g[i])/period; al=(al*(period-1)+l[i])/period
            r.append(100-(100/(1+ag/al)) if al else 100)
        return r

    @staticmethod
    def macd(prices):
        ef=QuantEngine.ema(prices,12); es=QuantEngine.ema(prices,26)
        ml=[f-s for f,s in zip(ef,es)]; sl=QuantEngine.ema(ml,9)
        return ml,sl,[m-s for m,s in zip(ml,sl)]

    @staticmethod
    def bollinger(prices, w=20, ns=2):
        u,m,lo=[],[],[]
        for i in range(len(prices)):
            if i<w-1: u.append(None);m.append(None);lo.append(None)
            else:
                s=prices[i-w+1:i+1]; mn=sum(s)/w
                std=statistics.stdev(s) if len(s)>1 else 0
                m.append(mn);u.append(mn+ns*std);lo.append(mn-ns*std)
        return u,m,lo

    @staticmethod
    def monte_carlo(price, vol, days=252, sims=500):
        dv=vol/math.sqrt(252); paths=[]; finals=[]
        for _ in range(sims):
            path=[price]; p=price
            for _ in range(days):
                p=p*math.exp(-0.5*dv**2+dv*random.gauss(0,1))
                path.append(round(p,4))
            paths.append(path); finals.append(p)
        finals.sort(); n=len(finals)
        return {"paths":[p for p in paths[:50]],"percentiles":{"p5":round(finals[int(n*.05)],2),"p25":round(finals[int(n*.25)],2),"p50":round(finals[int(n*.5)],2),"p75":round(finals[int(n*.75)],2),"p95":round(finals[int(n*.95)],2)},"mean":round(statistics.mean(finals),2),"std":round(statistics.stdev(finals),2),"current":price,"days":days}

    @staticmethod
    def risk_metrics(prices):
        rets=QuantEngine.returns(prices)
        if len(rets)<10: return {}
        ar=sum(rets)/len(rets)*252; av=statistics.stdev(rets)*math.sqrt(252)
        sharpe=(ar-0.05)/av if av else 0
        down=[r for r in rets if r<0]
        dv=statistics.stdev(down)*math.sqrt(252) if len(down)>1 else av
        sortino=(ar-0.05)/dv if dv else 0
        cum=[1]
        for r in rets: cum.append(cum[-1]*math.exp(r))
        pk=cum[0]; mdd=0
        for v in cum:
            if v>pk: pk=v
            dd=(pk-v)/pk
            if dd>mdd: mdd=dd
        sr=sorted(rets); var95=-sr[int(len(sr)*0.05)]*math.sqrt(252)
        tail=sr[:int(len(sr)*0.05)]
        cvar=-statistics.mean(tail)*math.sqrt(252) if tail else var95
        return {"annualReturn":round(ar*100,2),"annualVol":round(av*100,2),"sharpe":round(sharpe,3),"sortino":round(sortino,3),"maxDrawdown":round(mdd*100,2),"var95":round(var95*100,2),"cvar95":round(cvar*100,2)}

    @staticmethod
    def correlation(price_dict):
        tks=list(price_dict.keys()); rd={}; ml=float('inf')
        for t in tks:
            r=QuantEngine.returns(price_dict[t]); rd[t]=r; ml=min(ml,len(r))
        for t in tks: rd[t]=rd[t][-ml:]
        mat=[]
        for t1 in tks:
            row=[]
            for t2 in tks:
                if t1==t2: row.append(1.0)
                else:
                    r1,r2=rd[t1],rd[t2]
                    m1,m2=statistics.mean(r1),statistics.mean(r2)
                    s1,s2=statistics.stdev(r1),statistics.stdev(r2)
                    if s1>0 and s2>0:
                        cov=sum((a-m1)*(b-m2) for a,b in zip(r1,r2))/(len(r1)-1)
                        row.append(round(max(-1,min(1,cov/(s1*s2))),3))
                    else: row.append(0)
            mat.append(row)
        return {"tickers":tks,"matrix":mat}

    @staticmethod
    def fair_value_equity(info, price):
        eps=info.get("eps",0); fcf=info.get("fcf",0)
        pf=max(fcf,eps*0.7) if fcf else eps*0.7
        dcf=0
        for y in range(1,11):
            pf*=1.08; dcf+=pf/(1.10**y)
        term=pf*1.025/(0.10-0.025); dcf+=term/(1.10**10)
        shares=info.get("mcap",100)*1e9/price if price>0 else 1e9
        dcf_ps=dcf*1e9/shares
        spe={"Technology":30,"Financial":14,"Healthcare":22,"Energy":12,"Consumer Cyclical":25,"Industrials":20}
        pe_val=abs(eps)*spe.get(info.get("sector",""),20) if eps>0 else price
        fv=dcf_ps*0.5+pe_val*0.5; dev=((price-fv)/fv)*100
        return {"currentPrice":price,"fairValue":round(fv,2),"dcfValue":round(dcf_ps,2),"peValue":round(pe_val,2),"deviation":round(dev,2),"signal":"OVERVALUED" if dev>10 else "UNDERVALUED" if dev<-10 else "FAIR VALUE"}

    @staticmethod
    def fair_value_commodity(tk, price, vol):
        random.seed(hash(tk)%1000)
        hm=price*random.uniform(0.88,1.12)
        sd=MarketDataEngine.COMMODITY_SD.get(tk,{})
        s=sd.get("supply",100); d=sd.get("demand",100)
        prem=(d-s)/s*0.1 if s else 0
        fv=hm*(1+prem); dev=((price-fv)/fv)*100
        return {"currentPrice":price,"fairValue":round(fv,4),"deviation":round(dev,2),"signal":"OVERVALUED" if dev>10 else "UNDERVALUED" if dev<-10 else "FAIR VALUE"}

    @staticmethod
    def backtest(prices, fast=20, slow=50):
        if len(prices)<slow+1: return {"error":"Insufficient data"}
        sf=QuantEngine.ma(prices,fast); ss=QuantEngine.ma(prices,slow)
        pos=0; ep=0; trades=[]; eq=[100]; ce=100
        for i in range(slow,len(prices)):
            if sf[i] is None or ss[i] is None: eq.append(ce); continue
            if sf[i]>ss[i] and pos==0: pos=1; ep=prices[i]; trades.append({"type":"BUY","price":round(prices[i],4),"day":i})
            elif sf[i]<ss[i] and pos==1:
                ret=(prices[i]-ep)/ep; ce*=(1+ret); pos=0
                trades.append({"type":"SELL","price":round(prices[i],4),"day":i,"return":round(ret*100,2)})
            eq.append(round(ce,2))
        if pos==1:
            ret=(prices[-1]-ep)/ep; ce*=(1+ret)
            trades.append({"type":"SELL","price":round(prices[-1],4),"day":len(prices)-1,"return":round(ret*100,2)})
        tr=(ce-100); bh=(prices[-1]-prices[slow])/prices[slow]*100
        wins=[t for t in trades if t.get("return",0)>0]; losses=[t for t in trades if t.get("return",0)<0]
        return {"totalReturn":round(tr,2),"buyHoldReturn":round(bh,2),"numTrades":len([t for t in trades if t["type"]=="SELL"]),
                "winRate":round(len(wins)/(len(wins)+len(losses))*100,1) if (wins or losses) else 0,
                "equity":eq[-100:],"trades":trades[-20:]}

    @staticmethod
    def garch(prices):
        rets=QuantEngine.returns(prices)
        if len(rets)<10: return {"currentVol":0,"forecast":[]}
        omega=0.00001; alpha=0.1; beta=0.85
        var=statistics.variance(rets); vs=[var]
        for i in range(1,len(rets)):
            var=omega+alpha*rets[i-1]**2+beta*var; vs.append(var)
        fc=[]; v=vs[-1]
        for _ in range(30):
            v=omega+(alpha+beta)*v; fc.append(round(math.sqrt(v)*math.sqrt(252)*100,2))
        return {"currentVol":round(math.sqrt(vs[-1])*math.sqrt(252)*100,2),"historicalVols":[round(math.sqrt(x)*math.sqrt(252)*100,2) for x in vs[-60:]],"forecast":fc}

    def do_OPTIONS(self):
        self.send_response(200)
        for h,v in [("Access-Control-Allow-Origin","*"),("Access-Control-Allow-Methods","GET,POST,OPTIONS"),("Access-Control-Allow-Headers","Content-Type")]:
            self.send_header(h,v)
        self.end_headers()

    def _json(self, data):
        self.send_response(200); self.send_header("Content-Type","application/json"); self.send_header("Access-Control-Allow-Origin","*"); self.end_headers()
        self.wfile.write(json.dumps(data).encode())
